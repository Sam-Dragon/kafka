# Kafka Server

# Create Topic

kafka-topics.bat --list --zookeeper localhost:2181

# Produce a message
kafka-console-producer.bat --broker-list localhost:9092 --topic genericTopic

# Consume a message
kafka-console-consumer.bat --bootstrap-server localhost:9092 --topic genericTopic
